```swift
@MainActor
final class SmartPlaylistViewModel: Observable {
  // ... existing properties

  func validateSmartPlaylist(
    name: String,
    rules: SmartListRuleSet
  ) -> Result<Void, SmartPlaylistValidationError> {
    var errors: [String] = []

    if name.trimmingCharacters(in: .whitespaces).isEmpty {
      errors.append("Name cannot be empty")
    }

    if rules.rules.isEmpty {
      errors.append("At least one rule is required")
    }

    let ruleValidationResult = SmartListRuleValidator.validate(rules.rules)
    if case .failure(let ruleErrors) = ruleValidationResult {
      errors.append(contentsOf: ruleErrors.map(\.localizedDescription))
    }

    return errors.isEmpty ? .success(()) : .failure(.init(messages: errors))
  }

  func createSmartPlaylist(
    name: String,
    description: String,
    rules: SmartListRuleSet,
    sortBy: EpisodeSortBy,
    maxEpisodes: Int?,
    autoUpdate: Bool,
    refreshInterval: TimeInterval
  ) async throws {
    let validationResult = validateSmartPlaylist(name: name, rules: rules)
    if case .failure(let error) = validationResult {
      throw error
    }

    let newPlaylist = SmartEpisodeListV2(
      id: UUID().uuidString,
      name: name,
      description: description.isEmpty ? nil : description,
      rules: rules,
      sortBy: sortBy,
      maxEpisodes: maxEpisodes,
      autoUpdate: autoUpdate,
      refreshInterval: refreshInterval,
      createdAt: .now,
      lastUpdated: .now,
      isSystemGenerated: false
    )

    try await manager.createSmartPlaylist(newPlaylist)
    recordAnalyticsEvent(.playlistCreated, playlistId: newPlaylist.id)
    await reload()
  }
}
```

**Rationale**: Prevents invalid state from being persisted; provides detailed user feedback.

### Phase 6: Persistence & Analytics

**Step 10: Enhance UserDefaultsSmartPlaylistManager**

*File*: `Packages/Persistence/Sources/Persistence/UserDefaultsSmartPlaylistManager.swift`

```swift
final class UserDefaultsSmartPlaylistManager: SmartPlaylistManaging {
  private let lock = NSLock()

  func createSmartPlaylist(_ playlist: SmartEpisodeListV2) throws {
    lock.lock()
    defer { lock.unlock() }

    var custom = customCache
    custom.append(playlist)

    let encoder = JSONEncoder()
    encoder.dateEncodingStrategy = .iso8601

    let data = try encoder.encode(custom)
    userDefaults.set(data, forKey: "smart_episode_lists_v2_custom")

    customCache = custom
  }

  func updateSmartPlaylist(_ playlist: SmartEpisodeListV2) throws {
    lock.lock()
    defer { lock.unlock() }

    var custom = customCache
    guard let index = custom.firstIndex(where: { $0.id == playlist.id }) else {
      throw SmartPlaylistError.playlistNotFound(id: playlist.id)
    }

    let updated = playlist.withUpdatedAt(.now)
    custom[index] = updated

    let encoder = JSONEncoder()
    encoder.dateEncodingStrategy = .iso8601
    let data = try encoder.encode(custom)
    userDefaults.set(data, forKey: "smart_episode_lists_v2_custom")

    customCache = custom
  }
}
```

**Rationale**: Thread-safe persistence with audit timestamps.

### Test Reliability

**Strategies**:
- Use `.matching(identifier:).firstMatch` for all element discovery (duplicate-ID safe)
- Always scroll to target before interaction when SwiftUI list is involved
- Use `waitForExistence(timeout:)` for async UI updates
- Debounce preview updates to prevent flaky assertions on preview count
- Disable animations in test builds: `UIView.setAnimationsEnabled(false)`
- Isolate each test with fresh UserDefaults (via `IsolatedUITestCase`)

## Risk Mitigation

| Risk | Impact | Mitigation |
|------|--------|-----------|
| Live preview performance (100+ rules) | Medium | Debounce at 500ms, cache evaluation results |
| UI test flakiness (SwiftUI lazy lists) | High | Always scroll before interaction, use waitForExistence |
| Validation complexity (13 rule types) | Medium | Exhaustive testing of SmartListRuleValidator |
| Edit state corruption | High | Copy-on-write model, pre-flight validation |
| Persistence data migration | Low | Existing UserDefaults versioning in place |
| Concurrent rule updates | Low | NSLock in UserDefaultsSmartPlaylistManager |
